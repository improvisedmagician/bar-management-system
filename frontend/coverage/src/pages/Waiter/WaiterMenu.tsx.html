import { useState } from 'react';
import { useNavigate, useParams } from 'react-router-dom';
import { QrCode, CreditCard, Banknote, Printer } from 'lucide-react';
 
interface Product {
  id: number;
  name: string;
  price: number;
  category_id: number;
  type: 'Cozinha' | 'Bar';
}
 
interface OrderItem {
  cartId: number;
  product: Product;
  observations: string;
}
 
export default function WaiterMenu() {
  const navigate = useNavigate();
  const { tableId } = useParams();
  
  const [activeCategory, setActiveCategory] = useState<number | null>(null);
  const [cart, setCart] = useState<OrderItem[]>([]);
  
  // Consumo prévio da mesa (mock de itens que já foram pedidos antes)
  const [consumedItems] = useState([
    { id: 101, name: 'Chopp Pilsen 300ml', price: 12.0, quantity: 2 },
    { id: 102, name: 'Escondidinho', price: 48.0, quantity: 1 },
    { id: 103, name: 'Taxa de Serviço (10%)', price: 7.20, quantity: 1 }
  ]);
  
  // Modal states
  const [selectedProduct, setSelectedProduct] = useState<Product | null>(null);
  const [obsText, setObsText] = useState('');
  const [isCartOpen, setIsCartOpen] = useState(false);
  const [isCheckoutOpen, setIsCheckoutOpen] = useState(false);
  const [payments, setPayments] = useState<{method: string, amount: number}[]>([]);
 
  // Mock Data
  const categories = [
    { id: 1, name: 'Bebidas' },
    { id: 2, name: 'Petiscos' },
    { id: 3, name: 'Pratos' }
  ];
 
  const products: Product[] = [
    { id: 1, name: 'Chopp Pilsen 300ml', price: 12.0, category_id: 1, type: 'Bar' },
    { id: 2, name: 'Caipirinha de Limão', price: 25.0, category_id: 1, type: 'Bar' },
    { id: 3, name: 'Porção de Fritas', price: 35.0, category_id: 2, type: 'Cozinha' },
    { id: 4, name: 'Isca de Frango', price: 42.0, category_id: 2, type: 'Cozinha' },
    { id: 5, name: 'Escondidinho', price: 48.0, category_id: 3, type: 'Cozinha' },
  ];
 
  const filteredProducts = activeCategory 
    ? products.filter(p => p.category_id === activeCategory)
    : products;
 
  const handleAddToCart = () => {
    if (!selectedProduct) return;
    setCart([...cart, { cartId: Date.now(), product: selectedProduct, observations: obsText }]);
    setSelectedProduct(null);
    setObsText('');
  };
 
  const handleSendOrder = () => {
    if (cart.length === 0) return;
    const channel = new BroadcastChannel('kds_new_order');
    cart.forEach(item => {
      channel.postMessage({
        id: item.cartId, productName: item.product.name, tableNumber: Number(tableId),
        observations: item.observations || null, status: 'Na Fila', type: item.product.type, createdAt: Date.now()
      });
    });
    channel.close();
    setCart([]);
    setIsCartOpen(false);
    alert('Pedido enviado para a Cozinha/Bar com sucesso!');
  };
 
  const cartTotal = cart.reduce((acc, item) => acc + item.product.price, 0);
 
  // Lógica de Fechamento de Conta
  const totalConsumed = consumedItems.reduce((acc, item) => acc + (item.price * item.quantity), 0);
  const totalPaid = payments.reduce((acc, p) => acc + p.amount, 0);
  const remainingToPay = totalConsumed - totalPaid;
 
  const handleAddPayment = (method: string) => {
    if (remainingToPay > 0) {
      setPayments([...payments, { method, amount: remainingToPay }]);
    }
  };
 
  const handlePrint = () => {
    // Simularia comunicação via Web Serial API ou Bluetooth com a impressora térmica
    alert('Imprimindo comanda térmica na impressora do garçom...');
  };
 
  const handleFinishTable = () => {
    // Atualiza o mapa de mesas para Livre
    const saved = localStorage.getItem('bar_tables');
    if (saved) {
      const tables = JSON.parse(saved);
      const updatedTables = tables.map((t: any) => 
        t.number === Number(tableId) ? { ...t, status: 'Livre' } : t
      );
      localStorage.setItem('bar_tables', JSON.stringify(updatedTables));
    }
 
    alert('Mesa fechada com sucesso! Ela já consta como Livre no mapa.');
    navigate('/waiter/dashboard');
  };
 
  return (
    <div className="flex flex-col min-h-screen bg-gray-50 pb-20 relative">
      {/* Header */}
      <header className="bg-white shadow-sm p-4 sticky top-0 z-10">
        <div className="flex justify-between items-center mb-4">
          <button 
            onClick={() => navigate(-1)}
            className="text-gray-600 font-bold px-3 py-2 bg-gray-100 rounded-lg active:bg-gray-200"
          >
            Voltar
          </button>
          <h1 className="text-xl font-black text-gray-800 tracking-tight">Mesa {tableId}</h1>
          <div className="flex gap-2">
            <button 
              onClick={() => setIsCheckoutOpen(true)}
              className="bg-orange-100 text-orange-600 px-3 py-2 rounded-lg font-bold shadow-sm active:bg-orange-200"
            >
              Conta
            </button>
            <button 
              onClick={() => setIsCartOpen(true)}
              className="bg-blue-600 text-white px-3 py-2 rounded-lg font-bold shadow-sm active:bg-blue-700 flex items-center gap-2"
            >
              <span className="bg-white text-blue-600 px-2 rounded-full text-sm">{cart.length}</span>
            </button>
          </div>
        </div>
      </header>
 
      {/* Categorias */}
      <div className="flex overflow-x-auto p-4 gap-3 no-scrollbar border-b border-gray-200 bg-white">
        <button 
          onClick={() => setActiveCategory(null)}
          className={`whitespace-nowrap px-6 py-2 rounded-full font-bold transition-colors ${activeCategory === null ? 'bg-gray-800 text-white' : 'bg-gray-100 text-gray-600'}`}
        >
          Todos
        </button>
        {categories.map(cat => (
          <button 
            key={cat.id}
            onClick={() => setActiveCategory(cat.id)}
            className={`whitespace-nowrap px-6 py-2 rounded-full font-bold transition-colors ${activeCategory === cat.id ? 'bg-gray-800 text-white' : 'bg-gray-100 text-gray-600'}`}
          >
            {cat.name}
          </button>
        ))}
      </div>
 
      {/* Lista de Produtos */}
      <main className="flex-1 p-4">
        <div className="grid grid-cols-1 gap-3">
          {filteredProducts.map(product => (
            <div key={product.id} className="bg-white p-4 rounded-2xl shadow-sm border border-gray-100 flex justify-between items-center">
              <div>
                <h3 className="font-bold text-gray-800 text-lg">{product.name}</h3>
                <p className="text-green-600 font-bold mt-1">R$ {product.price.toFixed(2)}</p>
              </div>
              <button 
                onClick={() => setSelectedProduct(product)}
                className="bg-gray-100 text-blue-600 w-12 h-12 rounded-xl text-3xl font-light flex items-center justify-center active:bg-blue-600 active:text-white transition-colors"
              >
                +
              </button>
            </div>
          ))}
        </div>
      </main>
 
      {/* MODAL: Adicionar Observação */}
      {selectedProduct && (
        <div className="fixed inset-0 bg-black/60 z-50 flex items-end justify-center p-4">
          <div className="bg-white w-full max-w-md rounded-3xl p-6 shadow-2xl animate-fade-in-up">
            <div className="flex justify-between items-center mb-6">
              <h2 className="text-xl font-bold text-gray-800">{selectedProduct.name}</h2>
              <button onClick={() => setSelectedProduct(null)} className="text-gray-400 text-2xl font-bold px-2">&times;</button>
            </div>
            
            <label className="block text-sm font-bold text-gray-600 mb-2">Observações</label>
            <textarea 
              value={obsText}
              onChange={(e) => setObsText(e.target.value)}
              placeholder="Ex: Sem cebola..."
              className="w-full bg-gray-100 border-none rounded-xl p-4 text-gray-800 focus:ring-2 focus:ring-blue-500 outline-none h-24 resize-none mb-4"
            ></textarea>
            
            <div className="flex gap-2 overflow-x-auto no-scrollbar mb-6 pb-2">
              <button onClick={() => setObsText(prev => prev + 'Sem gelo ')} className="whitespace-nowrap px-4 py-2 bg-gray-100 rounded-lg text-sm font-semibold text-gray-700">Sem Gelo</button>
              <button onClick={() => setObsText(prev => prev + 'Gelo e limão ')} className="whitespace-nowrap px-4 py-2 bg-gray-100 rounded-lg text-sm font-semibold text-gray-700">Gelo e Limão</button>
            </div>
 
            <button 
              onClick={handleAddToCart}
              className="w-full bg-blue-600 text-white font-black py-4 rounded-xl text-lg shadow-lg shadow-blue-600/30"
            >
              Confirmar Adição
            </button>
          </div>
        </div>
      )}
 
      {/* MODAL: Comanda de Novos Pedidos */}
      {isCartOpen && (
        <div className="fixed inset-0 bg-black/60 z-50 flex flex-col justify-end">
          <div className="bg-white w-full h-[85vh] rounded-t-3xl flex flex-col shadow-2xl">
            <div className="p-6 border-b flex justify-between items-center">
              <div>
                <h2 className="text-2xl font-black">Novos Pedidos</h2>
                <p className="text-gray-500">Mesa {tableId}</p>
              </div>
              <button onClick={() => setIsCartOpen(false)} className="bg-gray-100 w-10 h-10 rounded-full font-bold">&times;</button>
            </div>
            <div className="flex-1 overflow-y-auto p-6">
              {cart.map(item => (
                <div key={item.cartId} className="flex justify-between items-start border-b pb-4 mb-4">
                  <div>
                    <p className="font-bold text-lg">{item.product.name}</p>
                    {item.observations && <p className="text-sm text-red-500">Obs: {item.observations}</p>}
                  </div>
                  <div className="flex flex-col items-end">
                    <span className="font-bold">R$ {item.product.price.toFixed(2)}</span>
                    <button onClick={() => setCart(cart.filter(i => i.cartId !== item.cartId))} className="text-red-500 text-sm font-bold mt-2">Remover</button>
                  </div>
                </div>
              ))}
            </div>
            <div className="p-6 bg-gray-50 border-t">
              <button 
                onClick={handleSendOrder}
                disabled={cart.length === 0}
                className="w-full bg-green-500 text-white py-4 rounded-xl font-black text-xl disabled:bg-gray-300"
              >
                ENVIAR PEDIDOS
              </button>
            </div>
          </div>
        </div>
      )}
 
      {/* MODAL: Fechamento de Conta (Mobile POS) */}
      {isCheckoutOpen && (
        <div className="fixed inset-0 bg-black/60 z-50 flex flex-col justify-end">
          <div className="bg-white w-full h-[95vh] rounded-t-3xl flex flex-col shadow-2xl">
            
            <div className="p-6 border-b flex justify-between items-center bg-gray-50 rounded-t-3xl">
              <div>
                <h2 className="text-2xl font-black text-gray-800">Fechamento</h2>
                <p className="text-gray-500 font-medium">Mesa {tableId}</p>
              </div>
              <button onClick={() => setIsCheckoutOpen(false)} className="bg-gray-200 w-10 h-10 rounded-full font-bold text-gray-700">&times;</button>
            </div>
 
            <div className="flex-1 overflow-y-auto p-6 flex flex-col gap-6">
              {/* Resumo da Conta */}
              <div className="bg-gray-50 p-4 rounded-2xl border border-gray-100">
                <h3 className="font-bold text-gray-500 mb-4 uppercase text-sm">Itens Consumidos</h3>
                {consumedItems.map(item => (
                  <div key={item.id} className="flex justify-between items-center mb-2">
                    <span className="font-semibold text-gray-800">{item.quantity}x {item.name}</span>
                    <span className="text-gray-600 font-bold">R$ {(item.price * item.quantity).toFixed(2)}</span>
                  </div>
                ))}
                <div className="border-t border-gray-200 mt-4 pt-4 flex justify-between items-center text-xl font-black text-gray-800">
                  <span>Total</span>
                  <span>R$ {totalConsumed.toFixed(2)}</span>
                </div>
              </div>
 
              {/* Botões de Ação do Garçom */}
              <button onClick={handlePrint} className="w-full bg-gray-100 text-gray-700 py-3 rounded-xl font-bold flex items-center justify-center gap-2 active:bg-gray-200 border border-gray-200">
                <Printer size={20} />
                Imprimir Pré-Conta
              </button>
 
              {/* Pagamentos */}
              <div>
                <h3 className="font-bold text-gray-800 mb-3 text-lg">Receber Pagamento</h3>
                <div className="grid grid-cols-3 gap-3 mb-4">
                  <button onClick={() => handleAddPayment('PIX')} className="bg-white border-2 border-gray-200 py-4 rounded-xl flex flex-col items-center gap-2 active:border-teal-500 active:bg-teal-50">
                    <QrCode size={24} className="text-gray-600" />
                    <span className="font-bold text-xs text-gray-600">PIX</span>
                  </button>
                  <button onClick={() => handleAddPayment('Cartão')} className="bg-white border-2 border-gray-200 py-4 rounded-xl flex flex-col items-center gap-2 active:border-blue-500 active:bg-blue-50">
                    <CreditCard size={24} className="text-gray-600" />
                    <span className="font-bold text-xs text-gray-600">Cartão</span>
                  </button>
                  <button onClick={() => handleAddPayment('Dinheiro')} className="bg-white border-2 border-gray-200 py-4 rounded-xl flex flex-col items-center gap-2 active:border-green-500 active:bg-green-50">
                    <Banknote size={24} className="text-gray-600" />
                    <span className="font-bold text-xs text-gray-600">Espécie</span>
                  </button>
                </div>
 
                {payments.length > 0 && (
                  <div className="bg-white border border-gray-200 p-4 rounded-xl mt-4">
                    <div className="flex justify-between items-center mb-4">
                      <span className="text-sm font-bold text-gray-500">Pagamentos Adicionados</span>
                      <button onClick={() => setPayments([])} className="text-xs text-red-600 font-bold bg-red-50 px-3 py-1 rounded-md active:bg-red-100">Limpar Todos</button>
                    </div>
                    {payments.map((p, i) => (
                      <div key={i} className="flex justify-between items-center text-sm font-bold text-gray-800 mb-3 border-b border-gray-50 pb-3 last:border-0 last:pb-0 last:mb-0">
                        <span className="bg-gray-100 px-2 py-1 rounded text-gray-600">{p.method}</span>
                        <div className="flex items-center gap-4">
                          <span className="text-base">R$ {p.amount.toFixed(2)}</span>
                          <button 
                            onClick={() => setPayments(payments.filter((_, index) => index !== i))} 
                            className="bg-red-100 text-red-600 w-8 h-8 flex items-center justify-center rounded-full text-lg font-bold active:bg-red-200"
                          >
                            &times;
                          </button>
                        </div>
                      </div>
                    ))}
                  </div>
                )}
              </div>
            </div>
 
            {/* Rodapé do Fechamento */}
            <div className="p-6 bg-white border-t border-gray-100 shadow-[0_-10px_20px_rgba(0,0,0,0.05)]">
              <div className="flex justify-between items-center mb-4">
                <span className="text-gray-500 font-bold text-lg">Falta Receber</span>
                <span className={`text-2xl font-black ${remainingToPay > 0 ? 'text-orange-500' : 'text-green-500'}`}>
                  R$ {Math.max(0, remainingToPay).toFixed(2)}
                </span>
              </div>
              
              <button 
                onClick={handleFinishTable}
                disabled={remainingToPay > 0}
                className={`w-full py-4 rounded-xl font-black text-lg transition-all ${remainingToPay <= 0 ? 'bg-green-500 text-white shadow-xl shadow-green-500/30 active:scale-95' : 'bg-gray-200 text-gray-400 cursor-not-allowed'}`}
              >
                {remainingToPay <= 0 ? 'FINALIZAR MESA' : 'AGUARDANDO PAGAMENTO'}
              </button>
            </div>
 
          </div>
        </div>
      )}
    </div>
  );
}

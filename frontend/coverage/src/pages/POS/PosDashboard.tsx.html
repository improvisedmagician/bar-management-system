import { useState } from 'react';
import { CreditCard, Banknote, QrCode, Split, Calculator } from 'lucide-react';
 
interface PosItem {
  id: number;
  name: string;
  price: number;
  quantity: number;
}
 
export default function PosDashboard() {
  // Mock data for POS
  const tables = [
    { id: 2, number: 2, total: 185.00, status: 'Fechando' },
    { id: 5, number: 5, total: 340.50, status: 'Aberto' },
  ];
 
  const [selectedTable, setSelectedTable] = useState(2);
  const [splitType, setSplitType] = useState<'none' | 'equal' | 'items'>('none');
  const [peopleCount, setPeopleCount] = useState(2);
  
  const [items] = useState<PosItem[]>([
    { id: 1, name: 'Chopp Pilsen 300ml', price: 12.0, quantity: 4 },
    { id: 2, name: 'Porção de Fritas', price: 35.0, quantity: 1 },
    { id: 3, name: 'Caipirinha de Limão', price: 25.0, quantity: 2 },
    { id: 4, name: 'Isca de Frango', price: 42.0, quantity: 1 },
    { id: 5, name: 'Taxa de Serviço (10%)', price: 10.0, quantity: 1 },
  ]);
 
  const subtotal = items.reduce((acc, item) => acc + (item.price * item.quantity), 0);
  const total = subtotal;
 
  const [payments, setPayments] = useState<{method: string, amount: number}[]>([]);
  const totalPaid = payments.reduce((acc, p) => acc + p.amount, 0);
  const remaining = total - totalPaid;
 
  const handleAddPayment = (method: string) => {
    if (remaining > 0) {
      setPayments([...payments, { method, amount: remaining }]);
    }
  };
 
  const handleClearPayments = () => {
    setPayments([]);
  }
 
  return (
    <div className="flex h-screen bg-gray-100 overflow-hidden font-sans">
      {/* Sidebar - Open Tables */}
      <aside className="w-80 bg-white border-r border-gray-200 flex flex-col">
        <div className="p-4 bg-gray-800 text-white">
          <h2 className="text-xl font-bold">Frente de Caixa</h2>
          <p className="text-sm text-gray-300">Turno Aberto: Maria</p>
        </div>
        <div className="flex-1 overflow-y-auto p-4 space-y-3">
          <h3 className="text-sm font-bold text-gray-500 uppercase tracking-wider mb-2">Mesas Abertas</h3>
          {tables.map(t => (
            <div 
              key={t.id} 
              onClick={() => setSelectedTable(t.number)}
              className={`p-4 rounded-xl border-2 cursor-pointer transition-all ${selectedTable === t.number ? 'border-blue-500 bg-blue-50' : 'border-gray-200 hover:border-gray-300'}`}
            >
              <div className="flex justify-between items-center mb-2">
                <span className="font-bold text-lg text-gray-800">Mesa {t.number}</span>
                <span className={`text-xs px-2 py-1 rounded-full font-bold ${t.status === 'Fechando' ? 'bg-orange-100 text-orange-700' : 'bg-green-100 text-green-700'}`}>
                  {t.status}
                </span>
              </div>
              <p className="text-gray-600 font-medium">Total: <span className="text-green-600">R$ {t.total.toFixed(2)}</span></p>
            </div>
          ))}
        </div>
      </aside>
 
      {/* Main Content - Bill Details */}
      <main className="flex-1 flex flex-col bg-gray-50">
        <header className="bg-white p-6 shadow-sm border-b flex justify-between items-center">
          <div>
            <h1 className="text-2xl font-black text-gray-800">Fechamento: Mesa {selectedTable}</h1>
            <p className="text-gray-500 font-medium">Consumo detalhado e pagamentos</p>
          </div>
          <button className="bg-gray-200 text-gray-700 px-4 py-2 rounded-lg font-bold hover:bg-gray-300 transition-colors">
            Imprimir Comanda
          </button>
        </header>
 
        <div className="flex-1 p-6 flex gap-6 overflow-hidden">
          {/* Bill Items */}
          <div className="flex-1 bg-white rounded-2xl shadow-sm border border-gray-200 flex flex-col">
            <div className="p-4 border-b bg-gray-50 rounded-t-2xl">
              <h3 className="font-bold text-gray-700">Itens Consumidos</h3>
            </div>
            <div className="flex-1 overflow-y-auto p-4 space-y-3">
              {items.map(item => (
                <div key={item.id} className="flex justify-between items-center py-2 border-b border-gray-100 last:border-0">
                  <div>
                    <p className="font-bold text-gray-800">{item.name}</p>
                    <p className="text-sm text-gray-500">{item.quantity}x R$ {item.price.toFixed(2)}</p>
                  </div>
                  <span className="font-bold text-gray-700">R$ {(item.price * item.quantity).toFixed(2)}</span>
                </div>
              ))}
            </div>
            <div className="p-6 bg-gray-50 border-t rounded-b-2xl">
              <div className="flex justify-between items-center mb-2 text-lg text-gray-600 font-medium">
                <span>Subtotal</span>
                <span>R$ {subtotal.toFixed(2)}</span>
              </div>
              <div className="flex justify-between items-center text-3xl font-black text-gray-800">
                <span>Total</span>
                <span>R$ {total.toFixed(2)}</span>
              </div>
            </div>
          </div>
 
          {/* Payment Panel */}
          <div className="w-[450px] flex flex-col gap-4">
            
            {/* Split Options */}
            <div className="bg-white p-4 rounded-2xl shadow-sm border border-gray-200">
              <h3 className="font-bold text-gray-700 mb-3 flex items-center gap-2">
                <Split size={18}/> Divisão de Conta
              </h3>
              <div className="flex gap-2">
                <button onClick={() => setSplitType('none')} className={`flex-1 py-2 rounded-lg font-semibold text-sm ${splitType === 'none' ? 'bg-blue-600 text-white' : 'bg-gray-100 text-gray-600 hover:bg-gray-200'}`}>Sem Divisão</button>
                <button onClick={() => setSplitType('equal')} className={`flex-1 py-2 rounded-lg font-semibold text-sm ${splitType === 'equal' ? 'bg-blue-600 text-white' : 'bg-gray-100 text-gray-600 hover:bg-gray-200'}`}>Igualitária</button>
                <button onClick={() => setSplitType('items')} className={`flex-1 py-2 rounded-lg font-semibold text-sm ${splitType === 'items' ? 'bg-blue-600 text-white' : 'bg-gray-100 text-gray-600 hover:bg-gray-200'}`}>Por Itens</button>
              </div>
 
              {splitType === 'equal' && (
                <div className="mt-4 flex items-center justify-between bg-blue-50 p-4 rounded-xl border border-blue-100">
                  <div className="flex items-center gap-4">
                    <button onClick={() => setPeopleCount(Math.max(2, peopleCount - 1))} className="w-8 h-8 bg-white rounded-full font-bold shadow text-blue-600">-</button>
                    <span className="font-bold text-lg text-blue-900">{peopleCount} Pessoas</span>
                    <button onClick={() => setPeopleCount(peopleCount + 1)} className="w-8 h-8 bg-white rounded-full font-bold shadow text-blue-600">+</button>
                  </div>
                  <span className="font-black text-xl text-blue-700">R$ {(total / peopleCount).toFixed(2)}</span>
                </div>
              )}
            </div>
 
            {/* Payment Methods */}
            <div className="flex-1 bg-white p-6 rounded-2xl shadow-sm border border-gray-200 flex flex-col">
              <h3 className="font-bold text-gray-700 mb-4 flex items-center gap-2">
                <Calculator size={18}/> Pagamentos
              </h3>
              
              <div className="grid grid-cols-2 gap-3 mb-6">
                <button onClick={() => handleAddPayment('PIX')} className="bg-gray-50 border-2 border-gray-200 p-4 rounded-xl flex flex-col items-center gap-2 hover:border-teal-500 hover:text-teal-600 hover:bg-teal-50 transition-colors">
                  <QrCode size={24} />
                  <span className="font-bold">PIX</span>
                </button>
                <button onClick={() => handleAddPayment('Cartão')} className="bg-gray-50 border-2 border-gray-200 p-4 rounded-xl flex flex-col items-center gap-2 hover:border-blue-500 hover:text-blue-600 hover:bg-blue-50 transition-colors">
                  <CreditCard size={24} />
                  <span className="font-bold">Cartão</span>
                </button>
                <button onClick={() => handleAddPayment('Dinheiro')} className="bg-gray-50 border-2 border-gray-200 p-4 rounded-xl flex flex-col items-center gap-2 hover:border-green-500 hover:text-green-600 hover:bg-green-50 transition-colors col-span-2">
                  <Banknote size={24} />
                  <span className="font-bold">Dinheiro</span>
                </button>
              </div>
 
              <div className="flex-1 overflow-y-auto mb-4 border-t border-gray-100 pt-4 relative">
                {payments.length > 0 && (
                  <button onClick={handleClearPayments} className="absolute top-2 right-0 text-xs font-bold text-red-500 bg-red-50 px-2 py-1 rounded">
                    Limpar Pagamentos
                  </button>
                )}
                <div className="mt-8">
                  {payments.map((p, i) => (
                    <div key={i} className="flex justify-between items-center text-sm font-medium text-gray-600 mb-2">
                      <span className="bg-gray-100 px-2 py-1 rounded">{p.method}</span>
                      <span className="text-gray-800 font-bold">R$ {p.amount.toFixed(2)}</span>
                    </div>
                  ))}
                </div>
              </div>
 
              <div className="mt-auto">
                <div className="flex justify-between items-center mb-4 p-4 bg-gray-100 rounded-xl">
                  <span className="font-bold text-gray-600">Falta Pagar</span>
                  <span className={`font-black text-3xl ${remaining > 0 ? 'text-red-500' : 'text-green-500'}`}>R$ {Math.max(0, remaining).toFixed(2)}</span>
                </div>
                
                <button 
                  disabled={remaining > 0}
                  className={`w-full py-4 rounded-xl font-bold text-lg transition-all shadow-sm ${remaining <= 0 ? 'bg-green-500 text-white hover:bg-green-600 hover:shadow-md' : 'bg-gray-200 text-gray-400 cursor-not-allowed'}`}
                >
                  {remaining <= 0 ? 'FINALIZAR MESA' : 'AGUARDANDO PAGAMENTO'}
                </button>
              </div>
            </div>
 
          </div>
        </div>
      </main>
    </div>
  );
}
